📋 项目概述

目标: 制作可穿戴设备，自动记录一天中"妈妈"这个词被呼唤的次数。

硬件平台: 微雪 ESP32-S3 1.85寸开发板 (360×360圆屏, 板载I2S麦克风, CNC铝合金外壳)

🔧 硬件构成

组件 型号 价格
🧠 主控开发板 微雪 ESP32-S3 1.85寸圆屏 ~￥80-100
🎤 麦克风 板载I2S数字麦克风 已集成
📐 陀螺仪 QMI8658 六轴传感器 已集成
🔋 电池 3.7V 锂电池 500mAh ~￥15

⚡ 核心功能

  • 声音检测: I2S数字麦克风，能量阈值检测算法
  • 翻转静音: 陀螺仪检测屏幕朝下，自动暂停监听
  • 大屏显示: 360×360分辨率，数字更清晰
  • 即开即用: CNC外壳，无需DIY

💻 开发环境

  • Arduino IDE + ESP32 Board Package
  • TFT_eSPI (屏幕驱动, 配置为360×360)
  • ESP32 I2S Driver (数字麦克风)
  • Wire (I2C陀螺仪通信)

🔋 功耗优化

模式 电流 策略
正常监听 ~80mA 100Hz采样，屏幕常亮
低功耗模式 ~10mA 降低亮度，降低采样率

/**
 * 妈妈计数器 - Mama Counter v1.1.0
 * 基于 ESP32-S3 1.85寸圆形LCD (360×360)
 * 
 * 硬件: 微雪 ESP32-S3 1.85寸开发板
 *       - 板载I2S数字麦克风
 *       - 板载陀螺仪QMI8658 (翻转静音)
 * 
 * 版本: v1.1.0 (开箱即用版)
 */

#include <TFT_eSPI.h>
#include <driver/i2s.h>
#include <Wire.h>

// 引脚定义 (微雪 ESP32-S3 1.85寸)
#define TFT_CS      10
#define TFT_SCK     12
#define TFT_MOSI    11
#define TFT_DC      13
#define TFT_BL      42

#define I2S_MIC_SCK  4
#define I2S_MIC_WS   5
#define I2S_MIC_SD   6

#define I2C_SDA     15
#define I2C_SCL     16
#define BUTTON_1     0

// 屏幕参数
#define SCREEN_WIDTH   360
#define SCREEN_HEIGHT  360
#define CENTER_X       180
#define CENTER_Y       180

// 配置
#define SAMPLE_RATE       16000
#define ENERGY_THRESHOLD  5000
#define TRIGGER_COOLDOWN  2000

// 颜色
#define COLOR_BG      TFT_BLACK
#define COLOR_PRIMARY 0x07E0    // 绿色
#define COLOR_TEXT    TFT_WHITE
#define COLOR_GRAY    0x8410

// 变量
TFT_eSPI tft = TFT_eSPI();
uint32_t mamaCount = 0;
uint32_t lastTriggerTime = 0;
bool isListening = true;
bool isFlipped = false;
int32_t i2sBuffer[256];

void setup() {
    Serial.begin(115200);
    
    pinMode(BUTTON_1, INPUT_PULLUP);
    pinMode(TFT_BL, OUTPUT);
    
    initDisplay();
    initI2SMic();
    initGyroscope();
    
    showBootScreen();
    delay(2000);
    
    tft.fillScreen(COLOR_BG);
    drawStaticUI();
}

void initI2SMic() {
    i2s_config_t cfg = {
        .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX),
        .sample_rate = SAMPLE_RATE,
        .bits_per_sample = I2S_BITS_PER_SAMPLE_32BIT,
        .channel_format = I2S_CHANNEL_FMT_ONLY_LEFT,
        .communication_format = I2S_COMM_FORMAT_STAND_I2S,
        .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
        .dma_buf_count = 4,
        .dma_buf_len = 256
    };
    
    i2s_pin_config_t pins = {
        .bck_io_num = I2S_MIC_SCK,
        .ws_io_num = I2S_MIC_WS,
        .data_out_num = I2S_PIN_NO_CHANGE,
        .data_in_num = I2S_MIC_SD
    };
    
    i2s_driver_install(I2S_NUM_0, &cfg, 0, NULL);
    i2s_set_pin(I2S_NUM_0, &pins);
}

int readI2SMicrophone() {
    size_t bytes_read;
    i2s_read(I2S_NUM_0, i2sBuffer, sizeof(i2sBuffer), &bytes_read, portMAX_DELAY);
    
    int64_t energy = 0;
    int num = bytes_read / sizeof(int32_t);
    
    for (int i = 0; i < num; i++) {
        energy += abs(i2sBuffer[i] >> 8);
    }
    return energy / num;
}

void checkFlipState() {
    // 读取QMI8658 Z轴
    Wire.beginTransmission(0x6B);
    Wire.write(0x35);
    Wire.endTransmission(false);
    Wire.requestFrom(0x6B, 2);
    
    if (Wire.available() >= 2) {
        int16_t az = (Wire.read() << 8) | Wire.read();
        isFlipped = (az / 16384.0 < -0.7);
    }
}

void loop() {
    checkFlipState();
    
    int energy = readI2SMicrophone();
    updateWaveform(energy / 100);
    
    if (!isFlipped && isListening && energy > ENERGY_THRESHOLD) {
        if (millis() - lastTriggerTime > TRIGGER_COOLDOWN) {
            triggerMamaDetected();
        }
    }
    
    checkButtons();
    updateStatusDisplay();
    delay(10);
}

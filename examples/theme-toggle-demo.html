📖 如何在项目中使用主题切换

1. 基础结构

在 CSS 中定义两套变量，通过 body 的 class 切换：

/* 默认（深空） */ :root { --bg-primary: #0a0a0f; --text-primary: #ffffff; /* ... 其他变量 */ } /* 白昼主题 */ body.daylight { --bg-primary: #f5f5f7; --text-primary: #1a1a1a; /* ... 其他变量 */ }

2. 切换 JavaScript

function toggleTheme() { const body = document.body; const isDaylight = body.classList.toggle('daylight'); // 保存用户偏好 localStorage.setItem('theme', isDaylight ? 'daylight' : 'dark'); } // 页面加载时恢复主题 const savedTheme = localStorage.getItem('theme'); if (savedTheme === 'daylight') { document.body.classList.add('daylight'); }

4. 自动跟随系统

也可以自动跟随系统的深色/浅色模式：

const prefersDark = window.matchMedia('(prefers-color-scheme: dark)'); function setThemeBySystem(e) { if (e.matches) { document.body.classList.remove('daylight'); } else { document.body.classList.add('daylight'); } } prefersDark.addEventListener('change', setThemeBySystem); setThemeBySystem(prefersDark);
